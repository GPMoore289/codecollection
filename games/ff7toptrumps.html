Description: I was looking for a coding project which would incorporate file reading. I remembered Top Trumps from my childhood and figured it would be a perfect choice. And as for Final Fantasy VII... I was replaying it at the time and I had the characters on the brain. Definitely could have picked better variables for the def function though.

Text file: Final Fantasy VII characters and their stats

bank = ["S", "M", "H", "C", "A"]
bank2 = ["Yes", "No"]
played = []
pscore = 0
oscore = 0
    
def battle(v1, v2, v3, v4, v5, v6, v7, v8, v9):
    if v1 == v2:
        if v3 > v4:
            print(v5 + v9 + v6 + " beat " + v7 + v9 + v8 + "!\nYou won!\n")
            global pscore
            pscore += 1
        else:
            print(v7 + v9 + v8 + " beat " + v5 + v9 + v6 + "!\nYou lost!\n")
            global oscore
            oscore += 1
            
print("Welcome to Final Fantasy VII Top Trumps! You and your opponent will each draw a character card. Choose a stat and beat your opponent!")
while True:
    if len(played) >= 30:
        if pscore > oscore:
            print("You win the game!")
        else:
            print("Sorry, you lost the game!")
        go = input("Game over. Would you like to play again (Yes or No)? ")
        if go not in bank2:
            print("Please enter a valid response")
        if go == "Yes":
            del played[0:30]
            pscore = 0
            oscore = 0
            continue
        if go == "No":
            break
    from random import choice
    file = open("ff7.txt", "r")
    a = choice(list(file))
    file = open("ff7.txt", "r")
    c = choice(list(file))
    if a == c:
        continue
    if a in played:
        continue
    if c in played:
        continue
    played.append(a)
    played.append(c)
    while True:
        b = a.split(',')
        d = c.split(',')
        ps = int(b[1])
        pm = int(b[2])
        ph = int(b[3])
        pc = int(b[4])
        pa = int(b[5])
        os = int(d[1])
        om = int(d[2])
        oh = int(d[3])
        oc = int(d[4])
        oa = int(d[5])
        if len(played)%4 == 0:
            import random
            ostat = random.choice(bank)
            print("\nYour opponent is deciding...")
            import time
            time.sleep(3)            
            print("\nYour opponent used " + d[0])
            battle(ostat, "S", ps, os, b[0], b[1], d[0], d[1], "'s Strength of ")
            battle(ostat, "M", pm, om, b[0], b[2], d[0], d[2], "'s Magic of ")
            battle(ostat, "H", ph, oh, b[0], b[3], d[0], d[3], "'s Honour of ")
            battle(ostat, "C", pc, oc, b[0], b[4], d[0], d[4], "'s Control of ")
            battle(ostat, "A", pa, oa, b[0], b[5], d[0], d[5], "'s Age of ")
        else:
            print("\nName:     " + b[0] + "\n" + "Strength: " + b[1] + "\n" + "Magic:    " + b[2] + "\n" + "Honour:   " + b[3] + "\n" + "Control:  " + b[4] + "\n" + "Age:      " + b[5])
            stat = input("Please choose a stat. S for Strength, M for Magic, H for Honour, C for Control or A for Age: ")
            if stat == "quit":
                break
            if not (stat) in bank:
                print("\nThat's not one of the five stats\n")
                continue
            print("\nYour opponent used " + d[0])
            battle(stat, "S", ps, os, b[0], b[1], d[0], d[1], "'s Strength of ")
            battle(stat, "M", pm, om, b[0], b[2], d[0], d[2], "'s Magic of ")
            battle(stat, "H", ph, oh, b[0], b[3], d[0], d[3], "'s Honour of ")
            battle(stat, "C", pc, oc, b[0], b[4], d[0], d[4], "'s Control of ")
            battle(stat, "A", pa, oa, b[0], b[5], d[0], d[5], "'s Age of ")
        print("Your score is " + str(pscore) + "\nYour opponent's score is " + str(oscore) + "\n")
        import time
        time.sleep(3)
        break
