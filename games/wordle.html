Description: Remember how people couldn't get enough of Wordle back in 2022, but were limited to one puzzle a day? Well, I found a solution: reverse-engineer it! This is based on my old version of Mastermind (this was before I knew how to use the def function) and while messy, it does work.

Text file: list of every five-letter word

import copy
from random import choice
file = open("wordlist.txt", "r")
enter = choice(list(file))
code = str.strip(enter)
bank = ["a", "b", "c", "d", "e", "f", "g", "h", "i", "j", "k", "l", "m", "n", "o", "p", "q", "r", "s", "t", "u", "v", "w", "x", "y", "z"]
guess = 6
word = ["_", "_", "_", "_", "_"]
ntprst = []

print("Welcome to MOOREDLE. \nA letter will either be correct (displayed normally), present but in the wrong position (displayed with asterisks), or absent.\nEnter 'quit' to quit.\n",word[0], word[1], word[2], word[3], word[4], "\n")
		
while guess > 0:
	lett = input("Enter a five-letter word: ")
	if lett == "quit":
		break
	if lett == "give":
		print(code)
	if len(lett) != 5:
		print("Must have five letters")
		continue
	lett.split(" ")
	if not (lett[0] and lett[1] and lett[2] and lett[3] and lett[4]) in bank:
		print("All letters entered must be part of the English alphabet")
		continue

	ans = copy.copy(word)
	ncode = list(code.strip(" "))

#What to do with first guessed letter
	if lett[0] == ncode[0]:
		ans[0] = lett[0]
		ncode[0] = " "
#If the 1st guessed letter is equal to the 2nd actual letter and the 2nd guessed letter is not equal to the 2nd actual letter
#then the 1st letter in the displayed answer becomes the 1st guessed letter
	elif lett[0] == ncode[1] and lett[1] != ncode[1]:
		ans[0] = "*" + str(lett[0]) + "*"
		ncode[1] = " "
	elif lett[0] == ncode[2] and lett[2] != ncode[2]:
		ans[0] = "*" + str(lett[0]) + "*"
		ncode[2] = " "
	elif lett[0] == ncode[3] and lett[3] != ncode[3]:
		ans[0] = "*" + str(lett[0]) + "*"
		ncode[3] = " "
	elif lett[0] == ncode[4] and lett[4] != ncode[4]:
		ans[0] = "*" + str(lett[0]) + "*"
		ncode[4] = " "
	else:
		if lett[0] not in ntprst and lett[0] not in ncode:
			ntprst.append(lett[0])

	if lett[1] == ncode[1]:
		ans[1] = lett[1]
		ncode[1] = " "
	elif lett[1] == ncode[0] and lett[0] != ncode[0]:
		ans[1] = "*" + str(lett[1]) + "*"
		ncode[0] = " "
	elif lett[1] == ncode[2] and lett[2] != ncode[2]:
		ans[1] = "*" + str(lett[1]) + "*"
		ncode[2] = " "
	elif lett[1] == ncode[3] and lett[3] != ncode[3]:
		ans[1] = "*" + str(lett[1]) + "*"
		ncode[3] = " "
	elif lett[1] == ncode[4] and lett[4] != ncode[4]:
		ans[1] = "*" + str(lett[1]) + "*"
		ncode[4] = " "
	else:
		if lett[1] not in ntprst and lett[1] not in ncode:
			ntprst.append(lett[1])

	if lett[2] == ncode[2]:
		ans[2] = lett[2]
		ncode[2] = " "
	elif lett[2] == ncode[0] and lett[0] != ncode[0]:
		ans[2] = "*" + str(lett[2]) + "*"
		ncode[0] = " "
	elif lett[2] == ncode[1] and lett[1] != ncode[1]:
		ans[2] = "*" + str(lett[2]) + "*"
		ncode[1] = " "
	elif lett[2] == ncode[3] and lett[3] != ncode[3]:
		ans[2] = "*" + str(lett[2]) + "*"
		ncode[3] = " "
	elif lett[2] == ncode[4] and lett[4] != ncode[4]:
		ans[2] = "*" + str(lett[2]) + "*"
		ncode[4] = " "
	else:
		if lett[2] not in ntprst and lett[2] not in ncode:
			ntprst.append(lett[2])  

	if lett[3] == ncode[3]:
		ans[3] = lett[3]
		ncode[3] = " "
	elif lett[3] == ncode[0] and lett[0] != ncode[0]:
		ans[3] = "*" + str(lett[3]) + "*"
		ncode[0] = " "
	elif lett[3] == ncode[1] and lett[1] != ncode[1]:
		ans[3] = "*" + str(lett[3]) + "*"
		ncode[1] = " "
	elif lett[3] == ncode[2] and lett[2] != ncode[2]:
		ans[3] = "*" + str(lett[3]) + "*"
		ncode[2] = " "
	elif lett[3] == ncode[4] and lett[4] != ncode[4]:
		ans[3] = "*" + str(lett[3]) + "*"
		ncode[4] = " "
	else:
		if lett[3] not in ntprst and lett[3] not in ncode:
			ntprst.append(lett[3])

	if lett[4] == ncode[4]:
		ans[4] = lett[4]
		ncode[4] = " "
	elif lett[4] == ncode[0] and lett[0] != ncode[0]:
		ans[4] = "*" + str(lett[4]) + "*"
		ncode[0] = " "
	elif lett[4] == ncode[1] and lett[1] != ncode[1]:
		ans[4] = "*" + str(lett[4]) + "*"
		ncode[1] = " "
	elif lett[4] == ncode[2] and lett[2] != ncode[2]:
		ans[4] = "*" + str(lett[4]) + "*"
		ncode[2] = " "
	elif lett[4] == ncode[3] and lett[3] != ncode[3]:
		ans[4] = "*" + str(lett[4]) + "*"
		ncode[3] = " "
	else:
		if lett[4] not in ntprst and lett[4] not in ncode:
			ntprst.append(lett[4])
	   
	print(" ".join(ans))
	ntprst.sort()
	print("Letters not in word: " + str(" ".join(ntprst)))
	if lett == code:
		print("You won!")
		break
	else:
		guess += -1
		print("Guesses remaining: " + str(guess) + "\n")
		if guess < 1:
			print("You lose!\nAnswer was " + str(code))
			break
		continue
