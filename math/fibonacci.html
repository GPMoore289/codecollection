Fibonacci Calculator

    def fibonacci(n):
        count = 0
        a = 1
        b = 0
        while count < n:
            count +=1
            if count == 1:
                yield b
                continue
            if count == 2:
                yield a
                continue
            if count > 2:
                c = a + b
                yield c
                if count%2 == 1:
                    b = c
                if count%2 == 0:
                    a = c
                continue
