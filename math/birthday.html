Birthday Calculator

Original Creation: 17th December 2022

Uploaded: 25th Feb 2025

Status: Complete. Original code was written without AI.

Description: An exercise in error-handling and considering edge-cases such as leap years, this calculates the day of the week you were born on, after inputting your birthday. Only works with dates after 1800.

while True:
    bday = input("Please enter your birthday in the format DD/MM/YYYY:")
    bdaylist = bday.split("/")
    try:
        for i in bdaylist:
            i = int(i)
    except ValueError:
            continue
    bdaylist = list(map(int, bdaylist))
    if bdaylist[1] > 12 or bdaylist[1] < 1 or bdaylist[0] < 1:
        print("That's not a real date.")
        continue
    if bdaylist[1] in (4, 6, 9, 11) and bdaylist[0] >30:
        print("That's not a real date.")
        continue
    if bdaylist[1] in (1, 3, 5, 7, 8, 10, 12) and bdaylist[0] > 31:
        print("That's not a real date.")
        continue
    if bdaylist[1] == 2:
        if bdaylist[0] > 29:
            print("That's not a real date.")
            continue
        elif bdaylist[2] % 4 != 0 and bdaylist[0] == 29:
            print("That's not a real date.")
            continue
        elif bdaylist[2] % 4 == 0 and bdaylist[0] == 29:
            if bdaylist[2] % 100 == 0 and bdaylist[2] % 400 != 0:
                print("That's not a real date.")
                continue
    if bdaylist[2] < 1800:
        print("This calculator only handles dates from 1800 CE onwards.")
        continue

    week = ["Wednesday", "Thursday", "Friday", "Saturday", "Sunday", "Monday", "Tuesday"]

    regyears = bdaylist[2]-1800
    lpyears = 0
    for i in range(1800, (bdaylist[2])):
        if i % 4 == 0:
            if i % 400 == 0:
                lpyears +=1
            elif i % 100 !=0:
                lpyears += 1

    daynum = 0

    if bdaylist[1] > 1:
        daynum += 31
    if bdaylist[1] > 2:
        if bdaylist[2] % 4 == 0:
            if bdaylist[2] % 400 == 0:
                daynum +=29
            elif bdaylist[2] % 100 !=0:
                daynum += 29
            else:
                daynum += 28
    if bdaylist[1] > 3:
        daynum += 31
    if bdaylist[1] > 4:
        daynum += 30
    if bdaylist[1] > 5:
        daynum += 31
    if bdaylist[1] > 6:
        daynum += 30
    if bdaylist[1] > 7:
        daynum += 31
    if bdaylist[1] > 8:
        daynum += 31
    if bdaylist[1] > 9:
        daynum += 30
    if bdaylist[1] > 10:
        daynum += 31
    if bdaylist[1] > 11:
        daynum += 30

    days = bdaylist[0] + daynum + 365*regyears + lpyears -1

    print("You were born on a " + str(week[days % 7]) + "\n")
