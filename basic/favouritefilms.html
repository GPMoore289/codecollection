Favourite Films

movies = ['Labyrinth', 'Aliens', 'Gremlins 2', 'Hot Fuzz', 'Total Recall', 'Everything Everywhere All at Once', 'Waltz With Bashir', 'Deadpool', 'X-Men 2']
commandlist = ['add', 'remove', 'print']

while True:
    command = input("Enter 'add', 'remove' or 'print':").lower()
    if command not in commandlist:
        print("Not a valid entry.")
        continue
    if command != "print":
        film = input("Enter name of film:")

    if command == 'add':
        if film in movies:
            print("Film is already in list.")
            continue
        else:
            movies.append(film)
    if command == 'remove':
        if film not in movies:
            print("Film is not in list.")
            continue
        else:
            movies.remove(film)
    if command == 'print':
        movies.sort()
        for movie in movies:
            print(movie)
