Description: The focus here was being able to pass arguments from the command line into the script and have them be interpreted correctly. While not called for in the specification, I wanted the script to be able to handle any number of delete requests and handle them appropriately, hence the loop.

#!/bin/bash
# Created by Graham Moore, 000957294, 29th March 2025. This script and the text file contents are entirely my own work.

allargs="$*" #this interprets all of the arguments as a single phrase and allows it to be added to the data file as such. This eliminates the need for quotes when providing arguments
datafile="./grahamsthoughts.txt" #Remember, NO SPACES!
deleteflag="False"

if [ $# -eq 0 ] #If the number of arguments is exactly zero...
then
    cat -n ${datafile} #...then just read the data file. needs -n to print line numbers
else
    for arg in "$@" # allows multiple delete arguments to be submitted at once, and iterated through.
    do
        #echo "first output: ${arg}" #for testing
        if [[ "${arg}" == "-h" ]] #using -eq is only good for numbers. Need to use == for string comparisons
        then
            echo -e "\nIf script is run with no arguments, all contents will be displayed."
            echo "If script is run with argument '-h', help will be displayed."
            echo "If script is run with argument '-d linenumber', that line will be deleted."
            echo -e "Otherwise, all arguments will be added to the notes list.\n"
            continue
        elif [[ "${arg}" == "-d" ]]
        then
            #echo "delete arg detected" #for testing
            deleteflag="True"
                 #I tried using the shift command to get to the next argument, but it didn't work as I expected.
                #This is a less elegant, but conceptually easier solution that seems to work, as long as the arguments are entered correctly.
            continue
        elif [[ "${deleteflag}" == "True" ]]
        then
            #echo "second output: ${arg}" #for testing
            lineno="${arg}d" # concatenates the current argument (line number) with the suffix d, which is needed for the next command
            #echo "third output: ${lineno}" #for testing
            sed -i ${lineno} ${datafile} #stream editor. Edits (deletes) lineno from datafile
            deleteflag="False"
            continue
        else #if there are any other arguments (i.e. thoughts, notes) they will be added to this file
            echo ${allargs} >> ${datafile} #appends the output of the echo statement (the argument) to the end of grahamsthoughts.txt
        fi
    done
fi
