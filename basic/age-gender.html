Description: A straightforward program that reads data from a csv file and produces average ages for each gender.

CSV: CSV data file used for this program

import csv
import statistics

def genderprint(genderlist, gender):
    average = round(statistics.mean(genderlist), 1)
    size = len(genderlist)
    print(f"The average age of {gender} was {average}, based on a sample size of {size}.")

data = []
with open("ages.csv", "r") as file:
    csvread = csv.reader(file)
    for row in csvread:
        data.append(row)

male = []
female = []
non-binary = []

for item in data:
    if item[2] == "Male":
        male.append(int(item[3]))
    elif item[2] == "Female":
        female.append(int(item[3]))
    elif item[2] == "Non-binary":
        non-binary.append(int(item[3]))

genderprint(male, "males")
genderprint(female, "females")
genderprint(non-binary, "non-binary people")
