Inventory Manager

Original Creation: 5th December 2024

Uploaded: 25th Mar 2025

Status: Complete. Original code was written without AI.

Description: Basic programming challenge emphasizing use of menus and correct code structure. Make a program that allows item names, quantities and prices to be entered into a list, then displayed again on demand. 

inventory = [] #List that stores items

#Allows user to select one of two options. No arguments/parameters. Returns a string of a valid option, either "1", "2" or "3".
def optioncheck(): 
    while True:
        selectoptions = ["1", "2", "3"]
        option = input("Please select a number: (1) Add item to inventory. (2) Display inventory. (3) Quit.")
        if option not in selectoptions:
            print("Please enter a valid option: 1, 2 or 3.")
            continue
        return option

#Allows user to enter name of product. No arguments/parameters. Returns the name as a string with no limitations on characters or length.
def itemnamecheck(): 
    name = input("Please enter the item name:")
    return name

#Allows user to enter amount of product. No arguments/parameters. Returns the quantity as a positive integer.
def itemquantcheck():
    while True:
        quant = input("Please enter item quantity:")
        try:
            int(quant)
        except:
            print("Item quantity must be an integer.")
            continue
        quant = int(quant)
        if quant < 0:
            print("Quantity must be zero or more.")
            continue
        return quant

#Allows user to enter price of product. No arguments/parameters. Returns the price as a positive float.
def itempricecheck():
    while True:
        price = input("Please enter item price, number only:")
        try:
            float(price)
        except:
            print("Item quantity must be a float.")
            continue
        price = float(price)
        if price < 0:
            print("Price must be zero or more.")
            continue
        return price

#Prints content from the inventory. Takes the inventory list as a parameter. Does not return any value.
def inventorydisplay(inventory):
    for item in inventory:
        print("\n" + str(item[0]))
        if item[1] >= 10:
            print("In stock")
        elif 3 <= item[1] <= 9:
            print("Almost gone")
        else:
            print("Reorder now")
        print(item[2])

#Assembles functions into a program. Takes no parameters and returns no value.
def main():
    while True:
        option = optioncheck()
        
        if option == "1":
            itemname = itemnamecheck()
            itemquant = itemquantcheck()
            itemprice = itempricecheck()
            inventory.append([itemname, itemquant, itemprice])
            
        if option == "2":
            inventorydisplay(inventory)

        if option == "3":
            break

main()
