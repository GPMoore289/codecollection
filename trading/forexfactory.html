Forex Factory Web Scraper

Description: While coding trading programs, I wanted to see the effect of trading close in time to high impact market events, such as US Federal Reserve Bank announcements. This was my introduction to web scrapers and their efficiency in acquiring data that would take a human days. This scraped data from the Forex Factory calendar and saved it as comma- and semi-colon-separated text file. All information obtained is publically available.

Original Creation: 10th Nov 2023

Uploaded: 14th Jun 2025

Status: Complete. Original code was written without AI.

Supplemental: Text: High Impact Event archive

import signal
from datetime import datetime

from selenium.webdriver.chrome.service import Service
from selenium import webdriver

options = webdriver.ChromeOptions()
options.add_argument('--headless=new')
user_agent = 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/83.0.4103.116 Safari/537.36'
options.add_argument(f'user-agent={user_agent}')

service = Service(executable_path='./chromedriver.exe')

months = ["jan", "feb", "mar", "apr", "may", "jun", "jul", "aug", "sep", "oct", "nov", "dec"]

def ManualInterrupt():

    print("Manual Interruption")
    driver.quit()
    myfile.close()

def keyboardInterruptHandler(signal, frame):

    ManualInterrupt()
    
signal.signal(signal.SIGINT, keyboardInterruptHandler)

with open('HIEarchive.txt', mode='a') as myfile:

    for y in range(2007, 2024):
        for z in months:
            for x in range(1, 10):
                
                if x >= 31 and (z == "apr" or z == "jun" or z == "sep" or z == "nov"):
                    continue
                
                if x >= 30 and z == "feb":
                    continue
                
                if x == 29 and z== "feb" and y%4 != 0:
                    continue
                
                driver = webdriver.Chrome(options=options)
                driver.get("https://www.forexfactory.com/calendar?day=" + z + str(x) + "." + str(y))
                # driver.get("https://www.forexfactory.com/calendar?day=feb9.2007")
                text = driver.page_source
                
                driver.close()
                
                HIElist = [i for i in range(len(text)) if text.startswith('High Impact Expected', i)]
                BHlist = [i for i in range(len(text)) if text.startswith('Non-Economic', i)]
                for i in BHlist:
                    HIElist.append(i)
                HIElisttrue = []
                Timelist = []
                Timelisttrue = []
                Currencylist = []
                Namelist = []
                
                for i in HIElist:
                    tlcheck1 = text[i+23:i+32]
                    tlcheck2 = text[i+15:i+24]
                    if tlcheck1 == "timeLabel" or tlcheck2 == "timeLabel":
                        HIElisttrue.append(i)
                
                if(len(HIElisttrue)) == 0:
                    continue
                
                timelastindex = 0
                
                for i in HIElisttrue:
                    afterslice = text[i:]
                    actualfind = afterslice.find("actual")
                    timelabelfind = afterslice.find("timeLabel")
                    time = text[i+timelabelfind+12:i+actualfind-3]
                    if time == "Tentative":
                        continue
                    Timelist.append(time)
                    beforeslice = text[i-350:]
                    currencyfind = beforeslice.find("currency")
                    Currencylist.append(text[i-350+currencyfind+11:i-350+currencyfind+14])
                    namefind = beforeslice.find("name")
                    dlfind = beforeslice.find("dateline")
                    Namelist.append(text[i-350+namefind+7:i-350+dlfind-3])
                    
                for elem in Timelist:
                    if elem.find("Day") != -1:
                        timestring = str(y) + "." + str(months.index(z)+1) + "." + str(x)
                        Timelisttrue.append(datetime.strptime(timestring, '%Y.%m.%d'))
                    else:
                        shorttime = elem[0:-2]
                        if elem.find("pm") != -1:
                            colonindex = shorttime.find(":")
                            hours = int(shorttime[0:colonindex])
                            if hours != 12:
                                shorttime = str(hours+12)+shorttime[colonindex:]
                        timestring = str(y) + "." + str(months.index(z)+1) + "." + str(x) + " " + shorttime + ":00"
                        Timelisttrue.append(datetime.strptime(timestring, '%Y.%m.%d %H:%M:%S'))
                    
                print(Timelisttrue)
                
                for a in range(0, len(Timelisttrue)):
                    myfile.writelines(str(Timelisttrue[a])+","+Currencylist[a]+","+Namelist[a]+";")
                
                
                
driver.quit()
myfile.close()
